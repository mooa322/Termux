# Installation
- pkg update -y
- pkg upgrade -y
- pkg install mpv -y
- pkg install git -y
- git clone https://github.com/mooa322/Termux.git
- cd Termux
- chmod +x *
- sh install.sh

Now Type exit & restart Termux
